Quick Start Snippet

Get up and running in minutes with this basic configuration example.

javascript 
// Initialize the client with your configuration
import { createClient } from '@example/sdk';

const client = createClient({
  apiKey: 'your-api-key',
  region: 'us-east-1',
  retries: 3,
  timeout: 5000,
});

// Fetch data with built-in error handling
const response = await client.query({
  collection: 'users',
  filter: { active: true },
  limit: 25,
});

console.log(`Found ${response.total} users`);
